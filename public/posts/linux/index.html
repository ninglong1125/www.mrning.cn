Linux命令简单记录 #

创建用户并分配bash #

useradd <user> -s /bin/bash

更改用户密码 #

passwd <username>

更改文件(夹)用户权限 #

chown -R <username> <filename>

查询进程 #

ps -aux | grep <name>

查询端口 #

lsof -i :8080

查询内存 #

free -h

查询磁盘 #

df -h

查询当前文件夹占用空间 #

du -sh

统计文件夹下的文件夹数目 #

ls | wc -w

查看文件后几行内容 #

 tail -n 5 <filename>

删除某一目录下的所有特定类型文件 #

find <dirname> -name "*.ckpt" | xargs rm -rf

解压*.tar.gz文件 #

tar -xvzf *.tar.gz

解压*.tar.xz文件 #

tar -xvf *.tar.xz

将文件夹压缩成*.tar.gz #

tar -zcvf <dirname>.tar.gz <dirname>

将文件夹压缩成*.zip #

zip -q -r <dirname>.zip <dirname>
